Meta-analysis stone recurrence for thiazide diuretic

#Stone Recurrence meta-analysis
MetaRecurrence <-metabin(event.e=thiazide$recurrence_r1, n.e=thiazide$recurrence_n1, event.c= thiazide$recurrence_r2, n.c= thiazide$recurrence_n2,
                    studlab = thiazide$Study, method = "MH", sm = "RR", random=TRUE, common=FALSE, allstudies = TRUE, method.tau = "DL", 
                    RR.Cochrane = TRUE, Q.Cochrane = TRUE, prediction= TRUE)
